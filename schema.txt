create table examiners(id int AUTO_INCREMENT,
     created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
     email varchar(100),
     password varchar(100),
     ay varchar(5),
     semester varchar(20),
     branch varchar(100),
     division varchar(20),
     exam varchar(100),
     internal_name varchar(100),
     external_name varchar(100),
     acc_no varchar(100),
     mob_no varchar(100),
     ifsc_code varchar(100),
     bank_address varchar(100),
     branch_name varchar(100),
     no_of_student int(5),
     amount_per_student float,
      PRIMARY KEY (id)
);
